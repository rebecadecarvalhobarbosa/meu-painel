function escHtml(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')}

// ===== GOOGLE AUTH (Identity Services) =====
function initGoogleAuth() {
if (!GOOGLE_CLIENT_ID || GOOGLE_CLIENT_ID.includes('SEU_CLIENT')) return;
tokenClient = google.accounts.oauth2.initTokenClient({
client_id: GOOGLE_CLIENT_ID,
scope: 'https://www.googleapis.com/auth/calendar.readonly',
callback: async (resp) => {
if (resp.error) { showBanner('error','Erro ao conectar: ' + resp.error); return; }
accessToken = resp.access_token;
tokenExpiry = Date.now() + (resp.expires_in - 60) * 1000;
setGoogleConnected(true);
await syncGoogleCalendars();
},
});
}

function handleGoogleAuth() {
if (!tokenClient) { alert('Configure o GOOGLE_CLIENT_ID no arquivo primeiro.'); return; }
if (accessToken && Date.now() < tokenExpiry) {
// já conectado — desconectar
google.accounts.oauth2.revoke(accessToken, () => {
accessToken = null; setGoogleConnected(false);
showBanner('info','Desconectado do Google Agenda.');
});
} else {
tokenClient.requestAccessToken();
}
}

function setGoogleConnected(on) {
const btn = document.getElementById('google-btn');
const txt = document.getElementById('google-btn-text');
btn.classList.toggle('connected', on);
txt.textContent = on ? 'Google conectado ✓' : 'Conectar Google Agenda';
}

// ===== SINCRONIZAÇÃO =====
async function syncGoogleCalendars() {
if (!accessToken) return;
showBanner('syncing','<span class="spinner"></span>Sincronizando agendas...');

const now = new Date();
const timeMin = new Date(now.getFullYear(), now.getMonth() - 1, 1).toISOString();
const timeMax = new Date(now.getFullYear(), now.getMonth() + 4, 0).toISOString();

const cals = [
{ id: CALENDAR_WORK || 'primary', category: 'work' },
...(CALENDAR_FAMILY && !CALENDAR_FAMILY.includes('id_da_agenda') ? [{ id: CALENDAR_FAMILY, category: 'family' }] : [])
];

let imported = 0;
for (const cal of cals) {
try {
const url = `https://www.googleapis.com/calendar/v3/calendars/${encodeURIComponent(cal.id)}/events`
+ `?timeMin=${encodeURIComponent(timeMin)}`
+ `&timeMax=${encodeURIComponent(timeMax)}`
+ `&singleEvents=true&maxResults=500&orderBy=startTime`;

const res = await fetch(url, {
headers: { Authorization: 'Bearer ' + accessToken }
});

if (!res.ok) {
const err = await res.json();
console.warn('Erro agenda', cal.id, err.error?.message);
continue;
}

const data = await res.json();
(data.items || []).forEach(ev => {
if (!ev.summary) return;
const dateRaw = ev.start?.dateTime || ev.start?.date;
if (!dateRaw) return;
const dateStr = dateRaw.split('T')[0];
const timeStr = ev.start?.dateTime ? ev.start.dateTime.slice(11,16) : undefined;
const exists = events.find(e => e.googleId === ev.id);
if (!exists) {
events.push({
id: nextId++, type: 'event',
title: ev.summary, date: dateStr,
time: timeStr, category: cal.category,
googleId: ev.id
});
imported++;
}
});
} catch(e) {
console.warn('Falha ao buscar agenda:', cal.id, e);
}
}

saveData();
renderCalendar();
showBanner('connected', `✅ Google Agenda sincronizado! ${imported} evento${imported !== 1 ? 's' : ''} importado${imported !== 1 ? 's' : ''}.`);
}

function showBanner(type, msg) {
const b = document.getElementById('sync-banner');
b.className = 'sync-banner' + (type === 'connected' ? ' connected' : type === 'syncing' ? ' syncing' : '');
document.getElementById('sync-msg').innerHTML = msg;
}

// ===== CALENDÁRIO =====
function renderCalendar() {
document.getElementById('month-title').textContent = `${MONTHS[curMonth]} ${curYear}`;
const grid = document.getElementById('cal-grid');
grid.querySelectorAll('.cal-day').forEach(e => e.remove());

const firstDay = new Date(curYear, curMonth, 1).getDay();
const daysInMonth = new Date(curYear, curMonth + 1, 0).getDate();
const prevMonthDays = new Date(curYear, curMonth, 0).getDate();
const rows = Math.ceil((firstDay + daysInMonth) / 7);

for (let i = 0; i < rows * 7; i++) {
const cell = document.createElement('div');
cell.className = 'cal-day';
let dayNum, isOther = false, y = curYear, m = curMonth;
if (i < firstDay) {
dayNum = prevMonthDays - firstDay + i + 1; isOther = true;
m = curMonth - 1; if (m < 0) { m = 11; y = curYear - 1; }
} else if (i >= firstDay + daysInMonth) {
dayNum = i - firstDay - daysInMonth + 1; isOther = true;
m = curMonth + 1; if (m > 11) { m = 0; y = curYear + 1; }
} else { dayNum = i - firstDay + 1; }

if (isOther) cell.classList.add('other-month');
const key = dateKey(y, m, dayNum);
if (key === todayKey()) cell.classList.add('today');
if (key === selectedDayKey) cell.classList.add('selected');

const numDiv = document.createElement('div');
numDiv.className = 'day-num'; numDiv.textContent = dayNum;
cell.appendChild(numDiv);

const evDiv = document.createElement('div'); evDiv.className = 'day-events';
const dayEvs = events.filter(e => e.date === key);
dayEvs.filter(e => e.type === 'event').slice(0,2).forEach(ev => {
const p = document.createElement('div');
p.className = 'event-pill ' + (CATS[ev.category]?.pill || 'ev-work');
p.textContent = (ev.time ? ev.time.slice(0,5) + ' ' : '') + ev.title;
evDiv.appendChild(p);
});
cell.appendChild(evDiv);

const tasks = dayEvs.filter(e => e.type === 'task');
if (tasks.length > 0) {
const tc = document.createElement('div'); tc.className = 'day-task-count';
const done = tasks.filter(t => t.done).length;
tc.textContent = `${done}/${tasks.length} tarefa${tasks.length > 1 ? 's' : ''}`;
cell.appendChild(tc);
}
cell.onclick = () => { selectedDayKey = key; renderCalendar(); openDayDetail(key, dayNum, m, y); };
grid.appendChild(cell);
}
renderWeekTasks();
}

function renderWeekTasks() {
const wt = document.getElementById('week-tasks-list');
const d = new Date();
const start = new Date(d); start.setDate(d.getDate() - d.getDay());
const end = new Date(start); end.setDate(start.getDate() + 6);
const weekTasks = events.filter(e => {
if (e.type !== 'task') return false;
const ed = new Date(e.date + 'T12:00:00');
return ed >= start && ed <= end;
});
if (weekTasks.length === 0) {
wt.innerHTML = '<div class="empty" style="padding:10px 0">Nenhuma tarefa esta semana</div>'; return;
}
wt.innerHTML = '';
weekTasks.forEach(t => buildTaskItem(t, wt, () => { renderCalendar(); renderWeekTasks(); }));
}

function buildTaskItem(t, container, onToggle) {
const cat = CATS[t.category] || CATS.task;
const item = document.createElement('div'); item.className = 'task-item';
item.innerHTML = `
<input type="checkbox" class="task-check" data-id="${t.id}" ${t.done ? 'checked' : ''}>
<span class="task-text" style="${t.done ? 'text-decoration:line-through;opacity:.45' : ''}">${escHtml(t.title)}</span>
<span class="task-tag ${cat.tag}">${cat.label}</span>
<button class="del-btn" title="Excluir">✕</button>`;
item.querySelector('.task-check').addEventListener('change', function() {
toggleTask(t.id, this.checked);
if (onToggle) onToggle();
});
item.querySelector('.del-btn').addEventListener('click', function(e) {
e.stopPropagation();
deleteItem(t.id, onToggle);
});
container.appendChild(item);
}

function openDayDetail(key, dayNum, month, year) {
const dayEvs = events.filter(e => e.date === key);
const html = buildDayHTML(key, dayNum, MONTHS[month], dayEvs);
const sidebar = document.getElementById('day-detail-content');
sidebar.innerHTML = html;
wireChecks(sidebar, key, dayNum, month, year);
if (window.innerWidth <= 760) {
const mob = document.getElementById('modal-day-content');
mob.innerHTML = `<button class="btn" style="margin-bottom:14px" onclick="document.getElementById('modal-day').classList.remove('open')">‹ Fechar</button>` + html;
wireChecks(mob, key, dayNum, month, year);
document.getElementById('modal-day').classList.add('open');
}
}

function buildDayHTML(key, dayNum, mName, dayEvs) {
const evOnly = dayEvs.filter(e => e.type === 'event');
const tasks = dayEvs.filter(e => e.type === 'task');
let html = `<div class="day-detail-header">${dayNum} de ${mName}</div>`;
if (evOnly.length > 0) {
html += `<div class="detail-section"><h4>Compromissos</h4>`;
evOnly.forEach(ev => {
const cat = CATS[ev.category] || CATS.work;
html += `<div class="detail-event">
<div class="dot ${cat.dot}"></div>
<span style="font-size:13px;line-height:1.4">${ev.time ? `<strong>${ev.time.slice(0,5)}</strong> ` : ''}${escHtml(ev.title)}</span>
<button class="del-btn" data-id="${ev.id}" title="Excluir">✕</button>
</div>`;
});
html += `</div>`;
}
if (tasks.length > 0) {
html += `<div class="detail-section"><h4>Tarefas</h4>`;
tasks.forEach(t => {
const cat = CATS[t.category] || CATS.task;
html += `<div class="task-item">
<input type="checkbox" class="task-check day-chk" data-id="${t.id}" ${t.done ? 'checked' : ''}>
<span class="task-text" style="${t.done ? 'text-decoration:line-through;opacity:.45' : ''}">${escHtml(t.title)}</span>
<span class="task-tag ${cat.tag}">${cat.label}</span>
<button class="del-btn day-del" data-id="${t.id}" title="Excluir">✕</button>
</div>`;
});
html += `</div>`;
}
if (!evOnly.length && !tasks.length) html += `<div class="empty" style="margin-top:8px">Dia livre! 🎉</div>`;
html += `<div style="margin-top:10px"><span class="link-add" onclick="openAddModal('${key}')">+ Adicionar neste dia</span></div>`;
return html;
}

function wireChecks(container, key, dayNum, month, year) {
container.querySelectorAll('.day-chk').forEach(cb => {
cb.addEventListener('change', function() {
toggleTask(parseInt(this.dataset.id), this.checked);
renderCalendar(); openDayDetail(key, dayNum, month, year);
renderTasksView(); renderDoneView();
});
});
container.querySelectorAll('.day-del, .detail-event .del-btn').forEach(btn => {
btn.addEventListener('click', function(e) {
e.stopPropagation();
deleteItem(parseInt(this.dataset.id), () => openDayDetail(key, dayNum, month, year));
});
});
}

function toggleTask(id, val) {
const t = events.find(e => e.id === id); if (!t) return;
t.done = val;
if (val) doneArchive.push({title:t.title, date:t.date, category:t.category});
else {
const idx = doneArchive.map(d=>d.title+d.date).lastIndexOf(t.title+t.date);
if (idx >= 0) doneArchive.splice(idx, 1);
}
saveData();
}

function confirmDelete(msg, onConfirm) {
document.getElementById('confirm-msg').textContent = msg;
document.getElementById('confirm-delete').classList.add('open');
const btn = document.getElementById('confirm-yes');
const newBtn = btn.cloneNode(true);
btn.parentNode.replaceChild(newBtn, btn);
newBtn.addEventListener('click', () => {
document.getElementById('confirm-delete').classList.remove('open');
onConfirm();
});
}

function deleteItem(id, afterDelete) {
const item = events.find(e => e.id === id);
if (!item) return;
const label = item.type === 'task' ? 'tarefa' : 'evento';
confirmDelete(`Excluir ${label} "${item.title}"? Esta ação não pode ser desfeita.`, () => {
events = events.filter(e => e.id !== id);
// remove from done archive too if it was there
doneArchive = doneArchive.filter(d => !(d.title === item.title && d.date === item.date));
saveData();
renderCalendar();
renderTasksView();
renderDoneView();
if (afterDelete) afterDelete();
});
}

function renderTasksView() {
const ms = `${curYear}-${String(curMonth+1).padStart(2,'0')}`;
const mt = events.filter(e => e.type==='task' && e.date.startsWith(ms));
document.getElementById('stat-total').textContent = mt.length;
document.getElementById('stat-done').textContent = mt.filter(t=>t.done).length;
document.getElementById('stat-pend').textContent = mt.filter(t=>!t.done).length;
const list = document.getElementById('all-tasks-list');
if (!mt.length) { list.innerHTML='<div class="empty">Nenhuma tarefa neste mês.</div>'; return; }
list.innerHTML='';
[...mt].sort((a,b)=>a.done-b.done).forEach(t => {
const cat = CATS[t.category]||CATS.task;
const item = document.createElement('div'); item.className='task-item';
item.innerHTML=`
<input type="checkbox" class="task-check" data-id="${t.id}" ${t.done?'checked':''}>
<span class="task-text" style="${t.done?'text-decoration:line-through;opacity:.45':''}">
${escHtml(t.title)}<span style="font-size:11px;color:var(--text3);margin-left:4px">${t.date}</span></span>
<span class="task-tag ${cat.tag}">${cat.label}</span>
<button class="del-btn" title="Excluir">✕</button>`;
item.querySelector('.task-check').addEventListener('change',function(){
toggleTask(parseInt(this.dataset.id),this.checked);
renderTasksView();renderDoneView();renderCalendar();
});
item.querySelector('.del-btn').addEventListener('click',function(e){
e.stopPropagation();
deleteItem(t.id, ()=>{renderTasksView();renderDoneView();});
});
list.appendChild(item);
});
}

function renderDoneView() {
const list = document.getElementById('done-list');
if (!doneArchive.length) {
list.innerHTML='<div class="empty">Suas realizações aparecerão aqui! ✨</div>'; return;
}
const grouped={};
[...doneArchive].reverse().forEach(d=>{
const p=d.date.split('-');
const k=`${MONTHS[parseInt(p[1])-1]} ${p[0]}`;
if(!grouped[k])grouped[k]=[];
grouped[k].push(d);
});
list.innerHTML='';
Object.keys(grouped).forEach(month=>{
const sec=document.createElement('div');sec.className='done-month';
sec.innerHTML=`<h4>${month}</h4>`;
grouped[month].forEach(d=>{
const cat=CATS[d.category]||CATS.task;
const item=document.createElement('div');item.className='done-item';
item.innerHTML=`<div class="check-icon"><svg viewBox="0 0 10 10" fill="none"><path d="M1.5 5l2.5 2.5L8.5 2.5" stroke="white" stroke-width="1.5" stroke-linecap="round" stroke-linejoin="round"/></svg></div>
<span>${escHtml(d.title)}</span>
<span class="task-tag ${cat.tag}" style="margin-left:auto">${cat.label}</span>`;
sec.appendChild(item);
});
list.appendChild(sec);
});
}

function showView(v) {
['calendar','tasks','done'].forEach(x=>{
document.getElementById('view-'+x).style.display=x===v?'block':'none';
});
document.querySelectorAll('.tab').forEach((t,i)=>{
t.classList.toggle('active',['calendar','tasks','done'][i]===v);
});
if(v==='tasks')renderTasksView();
if(v==='done')renderDoneView();
}

function prevMonth(){curMonth--;if(curMonth<0){curMonth=11;curYear--;}selectedDayKey=null;renderCalendar();}
function nextMonth(){curMonth++;if(curMonth>11){curMonth=0;curYear++;}selectedDayKey=null;renderCalendar();}

document.getElementById('add-date').value=prefillDate||todayKey();
updateAddForm();
document.getElementById('modal-add').classList.add('open');
}

}

function saveItem(){

events.push({
id:nextId++,

}
